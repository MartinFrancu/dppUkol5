89  std::vector<std::string> parser::parse_option_list(const std::string &str)
90  {
91  using namespace string_utils;
92 
93  std::string searched = str;
94  std::vector<std::string> result;
95  char delim = ',';
96 
97  size_t pos = find_first_nonescaped(searched, ',');
98  if (pos == std::string::npos) {
99  // if no escaped strokes are present in given string,
100  // try to use colon
101  delim = ':';
102  }
103 
104  while (true) {
105  pos = find_first_nonescaped(searched, delim);
106 
107  // extract option value and process it
108  std::string value = searched.substr(0, pos);
109  value = left_trim(value);
110  // check if last escaped character is whitespace
111  size_t whitespace_pos = find_last_escaped(value, ' ');
112  value = right_trim(value);
113  if (whitespace_pos != std::string::npos) {
114  // last character is escaped whitespace
115  if (value.size() == whitespace_pos) {
116  value.push_back(' ');
117  }
118  }
119  // finally unescape
120  value = unescape(value);
121 
122  // save extracted and processed option value and cut searched string
123  result.push_back(value);
124 
125  if (pos == std::string::npos) {
126  // no delimiter found
127  break;
128  }
129  searched = searched.substr(pos + 1);
130  }
131 
132  return result;
133  }

136  const config &cfg, const section &last_section, std::vector<std::string> &option_val_list, size_t line_number)
137  {

140  for (auto &opt_value : option_val_list) {
141  if (starts_with(opt_value, "${") && ends_with(opt_value, "}")) {
142  std::string link = opt_value.substr(2, opt_value.length() - 3);
143  size_t delim = find_first_nonescaped(link, '#');
144 
145  // link always has to be in format "section#option"
146  // section and option cannot be empty
147  if (delim == std::string::npos || (delim + 1) == link.length()) {
148  throw parser_exception("Bad format of link on line " + std::to_string(line_number));
149  }
150 
151  std::string sect_link = link.substr(0, delim);
152  std::string opt_link = link.substr(delim + 1);
153 
154  if (sect_link.empty()) {
155  throw parser_exception(
156  "Section name in link cannot be empty on line " + std::to_string(line_number));
157  }
158 
159  // find section with name specifid in link
160  const section *selected_section = nullptr;
161  if (last_section.get_name() == sect_link) {
162  selected_section = &last_section;
163  } else if (cfg.contains(sect_link)) {
164  selected_section = &cfg[sect_link];
165  } else {
166  throw parser_exception("Bad link on line " + std::to_string(line_number));
167  }
168 
169  // from selected section take appropriate option and set its value to options list
170  if (selected_section->contains(opt_link)) {
171  opt_value = selected_section->operator[](opt_link).get<string_ini_t>();
172  } else {
173  throw parser_exception("Option name in link not found on line " + std::to_string(line_number));

193  std::string line;
194  size_t line_number = 0;
195

197  line_number++;
198 
199  // if there was comment delete it
200  line = delete_comment(line);
201  line = left_trim(line);
202 
203  if (line.empty()) { // empty line
204  continue;
205  }
206  else if (starts_with(line, "[")) { // start of section
207  line = right_trim(line);
208  if (ends_with(line, "]")) {
209  // empty section name cannot be present
210  if (line.length() == 2) {
211  throw parser_exception("Section name cannot be empty on line " + std::to_string(line_number));
212  }
213 
214  // if there is cached section, save it
215  if (last_section != nullptr) {
216  cfg.add_section(*last_section);
217  }
218 
219  // extract name and validate it and finally create section object
220  std::string sect_name = unescape(line.substr(1, line.length() - 2));
221  validate_identifier(sect_name, line_number);
222  last_section = std::make_shared<section>(sect_name);
223  } else {
224  throw parser_exception("Section not ended on line " + std::to_string(line_number));
225  }
226  } else { // option
227  size_t opt_delim = find_first_nonescaped(line, '=');
228  if (opt_delim == std::string::npos) {
229  throw parser_exception("Unknown element option expected on line " + std::to_string(line_number));
230  }
231 
232  // if there is no opened section, option has no parent section
233  if (last_section == nullptr) {
234  throw parser_exception("Option not in section on line " + std::to_string(line_number));
235  }
236 
237  // equals character was right at the end of line, should not be
238  if ((opt_delim + 1) == line.length()) {
239  throw parser_exception("Option value cannot be empty on line " + std::to_string(line_number));
240  }
241 
242  // retrieve option name and value from line
243  std::string option_name = unescape(trim(line.substr(0, opt_delim)));
244  std::string option_val = line.substr(opt_delim + 1);
245 
246  // validate option name
247  validate_identifier(option_name, line_number);
248 
249  if (option_name.empty()) {
250  throw parser_exception("Option name cannot be empty on line " + std::to_string(line_number));
251  }
252 
253  auto option_val_list = parse_option_list(option_val);
254  if (option_val_list.empty()) {
255  throw parser_exception("Option value cannot be empty on line " + std::to_string(line_number));
256  }
257 
258  handle_links(cfg, *last_section, option_val_list, line_number);
259 
260  // and finally create option and store it in current section
261  option opt(option_name, option_val_list);
262  last_section->add_option(opt);
263  }
264  }
265 
266  // if there is cached section we have to add it to created config too
267  if (last_section != nullptr) {
268  cfg.add_section(*last_section);
269  }
270 
271  return cfg;

398  void parser::internal_save(const config &cfg, const schema &schm, std::ostream &str)
399  {
400  for (auto &sect : cfg) {
401  bool contains = schm.contains(sect.get_name());
402  if (!contains) {
403  // write section which is not in schema
404  // if this happens we can safely write all section and its option to output
405  // we do not have to go through them and write their additional info
406  str << sect;
407  continue;
408  }
409 
410  // if schema contains section from config, write additional info and name first
411  auto &sect_schema = schm[sect.get_name()];
412  sect_schema.write_additional_info(str);
413  sect_schema.write_section_name(str);
414 
415  // go through options and write them to output with info from option_schema
416  for (auto &opt : sect) {
417  if (sect_schema.contains(opt.get_name())) {
418  // if option is in section_schema, then write additional info
419  sect_schema[opt.get_name()].write_additional_info(str);
420  }
421 
422  // write option name and value to output
423  str << opt;
424  }
425 
426  // get through option_schema in appropriate section_schema
427  for (size_t i = 0; i < sect_schema.size(); ++i) {
428  auto &opt_schema = sect_schema[i];
429  if (sect.contains(opt_schema.get_name())) {
430  // already written to output stream
431  continue;
432  }
433 
434  // option with this name does not exist in config,
435  // so write its option_schema interpretation
436  str << opt_schema;
437  }
438  }
