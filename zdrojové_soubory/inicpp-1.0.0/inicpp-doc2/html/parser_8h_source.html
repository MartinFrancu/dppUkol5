120  {
121  using namespace string_utils;
122 
123  // repeat until there is a regular line, end of stack or exception
124  do {
125  do {
126  // check whether we run out of resources
127  if (stack_.empty()) {
128  return false;
129  }
130 
131  // try to read line from the resource
132  if (get_line(stack_.top(), out_string)) {
133  break;
134  }
135 
136  // no more lines in the current resource, get next
137  stack_.pop();
138  } while (true);
139 
140  // ignore whitespaces at the beginning
141  std::string line = left_trim(out_string);
142  if (starts_with(line, "#include")) {
143  std::string resource_name = trim(line.substr(8)); // remove "#include "
144 
145  // retrieve the new current resource
146  stack_.push(get_resource(resource_name));
147  }
148  else {
149  break;
150  }
151  } while (true);
