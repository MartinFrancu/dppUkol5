112  std::vector<std::string> parser::parse_option_list(const std::string &str)
113  {
114  using namespace string_utils;
115 
116  std::string searched = str;
117  std::vector<std::string> result;
118  char delim = ',';
119 
120  size_t pos = find_first_nonescaped(searched, ',');
121  if (pos == std::string::npos) {
122  // if no escaped strokes are present in given string,
123  // try to use colon
124  delim = ':';
125  }
126 
127  while (true) {
128  pos = find_first_nonescaped(searched, delim);
129 
130  // extract option value and process it
131  std::string value = searched.substr(0, pos);
132  value = left_trim(value);
133  // check if last escaped character is whitespace
134  size_t whitespace_pos = find_last_escaped(value, ' ');
135  value = right_trim(value);
136  if (whitespace_pos != std::string::npos) {
137  // last character is escaped whitespace
138  if (value.size() == whitespace_pos) {
139  value.push_back(' ');
140  }
141  }
142  // finally unescape
143  value = unescape(value);
144 
145  // save extracted and processed option value and cut searched string
146  result.push_back(value);
147 
148  if (pos == std::string::npos) {
149  // no delimiter found
150  break;
151  }
152  searched = searched.substr(pos + 1);
153  }
154 
155  return result;

158  void parser::handle_links(
159  const config &cfg, const section &last_section, std::vector<std::string> &option_val_list, size_t line_number)
160  {
161  using namespace string_utils;
162 
163  for (auto &opt_value : option_val_list) {
164  if (starts_with(opt_value, "${") && ends_with(opt_value, "}")) {
165  std::string link = opt_value.substr(2, opt_value.length() - 3);
166  size_t delim = find_first_nonescaped(link, '#');
167 
168  // link always has to be in format "section#option"
169  // section and option cannot be empty
170  if (delim == std::string::npos || (delim + 1) == link.length()) {
171  throw parser_exception("Bad format of link on line " + std::to_string(line_number));
172  }
173 
174  std::string sect_link = link.substr(0, delim);
175  std::string opt_link = link.substr(delim + 1);
176 
177  if (sect_link.empty()) {
178  throw parser_exception(
179  "Section name in link cannot be empty on line " + std::to_string(line_number));
180  }
181 
182  // find section with name specifid in link
183  const section *selected_section = nullptr;
184  if (last_section.get_name() == sect_link) {
185  selected_section = &last_section;
186  } else if (cfg.contains(sect_link)) {
187  selected_section = &cfg[sect_link];
188  } else {
189  throw parser_exception("Bad link on line " + std::to_string(line_number));
190  }
191 
192  // from selected section take appropriate option and set its value to options list
193  if (selected_section->contains(opt_link)) {
194  opt_value = selected_section->operator[](opt_link).get<string_ini_t>();
195  } else {
196  throw parser_exception("Option name in link not found on line " + std::to_string(line_number));
197  }
198  }
199  }

210  config parser::internal_load(resource &res)
211  {
212  using namespace string_utils;
213 
214  config cfg;
215  std::shared_ptr<section> last_section = nullptr;
216  std::string line;
217  size_t line_number = 0;
218 
219  while (res.get_line(line)) {
220  line_number++;
221 
222  // if there was comment delete it
223  line = delete_comment(line);
224  line = left_trim(line);
225 
226  if (line.empty()) { // empty line
227  continue;
228  }
229  else if (starts_with(line, "[")) { // start of section
230  line = right_trim(line);
231  if (ends_with(line, "]")) {
232  // empty section name cannot be present
233  if (line.length() == 2) {
234  throw parser_exception("Section name cannot be empty on line " + std::to_string(line_number));
235  }
236 
237  // if there is cached section, save it
238  if (last_section != nullptr) {
239  cfg.add_section(*last_section);
240  }
241 
242  // extract name and validate it and finally create section object
243  std::string sect_name = unescape(line.substr(1, line.length() - 2));
244  validate_identifier(sect_name, line_number);
245  last_section = std::make_shared<section>(sect_name);
246  } else {
247  throw parser_exception("Section not ended on line " + std::to_string(line_number));
248  }
249  } else { // option
250  size_t opt_delim = find_first_nonescaped(line, '=');
251  if (opt_delim == std::string::npos) {
252  throw parser_exception("Unknown element option expected on line " + std::to_string(line_number));
253  }
254 
255  // if there is no opened section, option has no parent section
256  if (last_section == nullptr) {
257  throw parser_exception("Option not in section on line " + std::to_string(line_number));
258  }
259 
260  // equals character was right at the end of line, should not be
261  if ((opt_delim + 1) == line.length()) {
262  throw parser_exception("Option value cannot be empty on line " + std::to_string(line_number));
263  }
264 
265  // retrieve option name and value from line
266  std::string option_name = unescape(trim(line.substr(0, opt_delim)));
267  std::string option_val = line.substr(opt_delim + 1);
268 
269  // validate option name
270  validate_identifier(option_name, line_number);
271 
272  if (option_name.empty()) {
273  throw parser_exception("Option name cannot be empty on line " + std::to_string(line_number));
274  }
275 
276  auto option_val_list = parse_option_list(option_val);
277  if (option_val_list.empty()) {
278  throw parser_exception("Option value cannot be empty on line " + std::to_string(line_number));
279  }
280 
281  handle_links(cfg, *last_section, option_val_list, line_number);
282 
283  // and finally create option and store it in current section
284  option opt(option_name, option_val_list);
285  last_section->add_option(opt);
286  }
287  }
288 
289  // if there is cached section we have to add it to created config too
290  if (last_section != nullptr) {
291  cfg.add_section(*last_section);
292  }
293 
294  return cfg;
295  }

297  void parser::internal_save(const config &cfg, const schema &schm, std::ostream &str)
298  {
299  for (auto &sect : cfg) {
300  bool contains = schm.contains(sect.get_name());
301  if (!contains) {
302  // write section which is not in schema
303  // if this happens we can safely write all section and its option to output
304  // we do not have to go through them and write their additional info
305  str << sect;
306  continue;
307  }
308 
309  // if schema contains section from config, write additional info and name first
310  auto &sect_schema = schm[sect.get_name()];
311  sect_schema.write_additional_info(str);
312  sect_schema.write_section_name(str);
313 
314  // go through options and write them to output with info from option_schema
315  for (auto &opt : sect) {
316  if (sect_schema.contains(opt.get_name())) {
317  // if option is in section_schema, then write additional info
318  sect_schema[opt.get_name()].write_additional_info(str);
319  }
320 
321  // write option name and value to output
322  str << opt;
323  }
324 
325  // get through option_schema in appropriate section_schema
326  for (size_t i = 0; i < sect_schema.size(); ++i) {
327  auto &opt_schema = sect_schema[i];
328  if (sect.contains(opt_schema.get_name())) {
329  // already written to output stream
330  continue;
331  }
332 
333  // option with this name does not exist in config,
334  // so write its option_schema interpretation
335  str << opt_schema;
336  }
337  }
338  }
